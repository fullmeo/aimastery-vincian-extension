### 1. Dependency Injection Container

**File**: `src/core/di/ServiceContainer.ts` (~200 lines)

```typescript
/**
 * Central dependency injection container for v8.0
 * Replaces all Singleton patterns from v7.x
 */
export class ServiceContainer {
  private services: Map<string, any> = new Map();
  private factories: Map<string, () => any> = new Map();

  /**
   * Register a singleton service
   */
  registerSingleton<T>(key: string, instance: T): void {
    this.services.set(key, instance);
  }

  /**
   * Register a factory for lazy instantiation
   */
  registerFactory<T>(key: string, factory: () => T): void {
    this.factories.set(key, factory);
  }

  /**
   * Resolve a service (lazy instantiation)
   */
  resolve<T>(key: string): T {
    if (this.services.has(key)) {
      return this.services.get(key);
    }

    if (this.factories.has(key)) {
      const instance = this.factories.get(key)!();
      this.services.set(key, instance);
      return instance;
    }

    throw new Error(`Service not found: ${key}`);
  }

  /**
   * Clear all services (for testing)
   */
  clear(): void {
    this.services.clear();
    this.factories.clear();
  }
}

// Global container instance
export const container = new ServiceContainer();
```

**Usage Example**:
```typescript
// Register services at startup
container.registerSingleton('logger', new Logger());
container.registerFactory('codeAnalyzer', () => new CodeAnalyzer(
  container.resolve('logger'),
  container.resolve('cache')
));

// Resolve anywhere
const analyzer = container.resolve<CodeAnalyzer>('codeAnalyzer');
```

import * as vscode from 'vscode';

/**
 * Standard interface all features must implement
 */
export interface IFeature {
  readonly id: string;
  readonly name: string;
  readonly description: string;
  readonly version: string;

  activate(context: vscode.ExtensionContext): Promise<void>;
  deactivate(): Promise<void>;
  isEnabled(): boolean;
}

/**
 * Base class providing common feature functionality
 */
export abstract class BaseFeature implements IFeature {
  protected context!: vscode.ExtensionContext;
  protected disposables: vscode.Disposable[] = [];

  abstract readonly id: string;
  abstract readonly name: string;
  abstract readonly description: string;
  abstract readonly version: string;

  constructor(
    protected container: ServiceContainer,
    protected config: ConfigurationManager
  ) {}

  async activate(context: vscode.ExtensionContext): Promise<void> {
    this.context = context;

    if (!this.isEnabled()) {
      console.log(`Feature ${this.id} is disabled`);
      return;
    }

    await this.onActivate();
    this.registerCommands();
    this.registerProviders();
  }

  async deactivate(): Promise<void> {
    await this.onDeactivate();
    this.disposables.forEach(d => d.dispose());
    this.disposables = [];
  }

  isEnabled(): boolean {
    return this.config.get(`features.${this.id}.enabled`, true);
  }

  protected abstract onActivate(): Promise<void>;
  protected abstract onDeactivate(): Promise<void>;
  protected abstract registerCommands(): void;
  protected abstract registerProviders(): void;

  protected registerCommand(
    command: string,
    handler: (...args: any[]) => any
  ): void {
    const disposable = vscode.commands.registerCommand(command, handler);
    this.disposables.push(disposable);
    this.context.subscriptions.push(disposable);
  }
}

### 3. Worker Thread Pool for AST Parsing

**File**: `src/core/workers/WorkerPool.ts` (~300 lines)

```typescript
import { Worker } from 'worker_threads';
import * as path from 'path';

export interface WorkerTask<T> {
  id: string;
  type: string;
  data: any;
  resolve: (value: T) => void;
  reject: (error: Error) => void;
}

/**
 * Worker thread pool for CPU-intensive operations
 * Prevents UI blocking during AST parsing
 */
export class WorkerPool {
  private workers: Worker[] = [];
  private availableWorkers: Worker[] = [];
  private taskQueue: WorkerTask<any>[] = [];
  private activeTasks: Map<string, WorkerTask<any>> = new Map();

  constructor(
    private workerScript: string,
    private poolSize: number = 4 // CPU cores - 1
  ) {
    this.initialize();
  }

  private initialize(): void {
    for (let i = 0; i < this.poolSize; i++) {
      const worker = new Worker(this.workerScript);

      worker.on('message', (message) => {
        this.handleWorkerMessage(worker, message);
      });

      worker.on('error', (error) => {
        console.error('Worker error:', error);
      });

      this.workers.push(worker);
      this.availableWorkers.push(worker);
    }
  }

  /**
   * Execute a task in the worker pool
   */
  execute<T>(type: string, data: any): Promise<T> {
    return new Promise((resolve, reject) => {
      const task: WorkerTask<T> = {
        id: `task_${Date.now()}_${Math.random()}`,
        type,
        data,
        resolve,
        reject
      };

      this.taskQueue.push(task);
      this.processQueue();
    });
  }

  private processQueue(): void {
    while (this.taskQueue.length > 0 && this.availableWorkers.length > 0) {
      const task = this.taskQueue.shift()!;
      const worker = this.availableWorkers.shift()!;

      this.activeTasks.set(task.id, task);

      worker.postMessage({
        id: task.id,
        type: task.type,
        data: task.data
      });
    }
  }

  private handleWorkerMessage(worker: Worker, message: any): void {
    const task = this.activeTasks.get(message.id);

    if (!task) {
      console.error('Unknown task ID:', message.id);
      return;
    }

    this.activeTasks.delete(message.id);
    this.availableWorkers.push(worker);

    if (message.error) {
      task.reject(new Error(message.error));
    } else {
      task.resolve(message.result);
    }

    this.processQueue();
  }

  /**
   * Terminate all workers
   */
  async terminate(): Promise<void> {
    await Promise.all(this.workers.map(w => w.terminate()));
    this.workers = [];
    this.availableWorkers = [];
    this.taskQueue = [];
    this.activeTasks.clear();
  }

  /**
   * Get pool statistics
   */
  getStats() {
    return {
      totalWorkers: this.workers.length,
      availableWorkers: this.availableWorkers.length,
      activeWorkers: this.workers.length - this.availableWorkers.length,
      queuedTasks: this.taskQueue.length,
      activeTasks: this.activeTasks.size
    };
  }
}
```

import * as vscode from 'vscode';
import { IFeature } from './BaseFeature';

/**
 * Central registry for all extension features
 * Handles feature lifecycle and dependency resolution
 */
export class FeatureRegistry {
  private features: Map<string, IFeature> = new Map();
  private activatedFeatures: Set<string> = new Set();

  /**
   * Register a feature
   */
  register(feature: IFeature): void {
    if (this.features.has(feature.id)) {
      throw new Error(`Feature already registered: ${feature.id}`);
    }

    this.features.set(feature.id, feature);
    console.log(`Registered feature: ${feature.name} (${feature.id})`);
  }

  /**
   * Activate all enabled features
   */
  async activateAll(context: vscode.ExtensionContext): Promise<void> {
    const startTime = Date.now();
    const activations: Promise<void>[] = [];

    for (const [id, feature] of this.features) {
      if (feature.isEnabled()) {
        activations.push(
          this.activateFeature(id, context)
            .catch(error => {
              console.error(`Failed to activate ${id}:`, error);
            })
        );
      }
    }

    await Promise.all(activations);

    const duration = Date.now() - startTime;
    console.log(`Activated ${this.activatedFeatures.size} features in ${duration}ms`);
  }

  /**
   * Activate a single feature
   */
  async activateFeature(
    featureId: string,
    context: vscode.ExtensionContext
  ): Promise<void> {
    const feature = this.features.get(featureId);

    if (!feature) {
      throw new Error(`Feature not found: ${featureId}`);
    }

    if (this.activatedFeatures.has(featureId)) {
      console.log(`Feature already activated: ${featureId}`);
      return;
    }

    const startTime = Date.now();
    await feature.activate(context);
    this.activatedFeatures.add(featureId);

    const duration = Date.now() - startTime;
    console.log(`‚úÖ ${feature.name} activated in ${duration}ms`);
  }

  /**
   * Deactivate a single feature
   */
  async deactivateFeature(featureId: string): Promise<void> {
    const feature = this.features.get(featureId);

    if (!feature) {
      throw new Error(`Feature not found: ${featureId}`);
    }

    if (!this.activatedFeatures.has(featureId)) {
      console.log(`Feature not activated: ${featureId}`);
      return;
    }

    await feature.deactivate();
    this.activatedFeatures.delete(featureId);
    console.log(`‚ùå ${feature.name} deactivated`);
  }

  /**
   * Deactivate all features
   */
  async deactivateAll(): Promise<void> {
    const deactivations: Promise<void>[] = [];

    for (const featureId of this.activatedFeatures) {
      deactivations.push(
        this.deactivateFeature(featureId)
          .catch(error => {
            console.error(`Failed to deactivate ${featureId}:`, error);
          })
      );
    }

    await Promise.all(deactivations);
    console.log('All features deactivated');
  }

  /**
   * Get feature by ID
   */
  get(featureId: string): IFeature | undefined {
    return this.features.get(featureId);
  }

  /**
   * Get all registered features
   */
  getAll(): IFeature[] {
    return Array.from(this.features.values());
  }

  /**
   * Get activated features
   */
  getActivated(): IFeature[] {
    return Array.from(this.activatedFeatures)
      .map(id => this.features.get(id))
      .filter((f): f is IFeature => f !== undefined);
  }
}

      return this.analyzeCurrentFile();

      return this.analyzeWorkspace();

    this.registerCommand('aimastery.autoFix', () => {
      return this.autoFix();
    });
  }

    // Tree provider registered in onActivate
  }

  private async analyzeCurrentFile(): Promise<void> {
    const editor = vscode.window.activeTextEditor;
    if (!editor) {
      vscode.window.showWarningMessage('No active editor');
      return;
    }

    const document = editor.document;
    const content = document.getText();

    // Check cache first
    const cached = this.cache.get(document.uri.fsPath, content);
    if (cached) {
      console.log('‚úÖ Cache HIT');
      this.showAnalysisResult(cached);
      return;
    }

    // Parse in worker thread (non-blocking)
    const result = await this.workerPool.execute('parseTypeScript', {
      content,
      options: { jsx: true }
    });

    // Cache result
    this.cache.set(document.uri.fsPath, content, result);

    this.showAnalysisResult(result);
  }

  private showAnalysisResult(result: any): void {
    // Display results in tree view
    this.treeProvider.setResults(result);
  }
}

      'aimasteryAIResults',
      'AI Query Results',
      vscode.ViewColumn.Two,
      {}
    );

    panel.webview.html = this.renderQueryResults(result);
  }

    const editor = vscode.window.activeTextEditor;
    if (!editor) return;

    const suggestions = await this.suggestionProvider.getSuggestionsForContext(
      editor.document,
      editor.selection.active
    );

    const items = suggestions.map(s => ({
      label: s.title,
      description: s.description,
      detail: s.code
    }));

    const selected = await vscode.window.showQuickPick(items);

    if (selected) {
      editor.edit(editBuilder => {
        editBuilder.insert(editor.selection.active, selected.detail);
      });
    }
